'TEST CASE 1: POSITIVE LOGIN TEST

'---Open page
SystemUtil.Run "C:\Program Files\Google\Chrome\Application\chrome.exe", “https://practicetestautomation.com/practice-test-login”

'---Type username student into Username field
Browser("Practice Test Automation").Page("Login").WebEdit("Username").Set "student"

'---Type password Password123 into Password field
Browser("Practice Test Automation").Page("Login").WebEdit("Password").Set "Password123"

'---Click Submit button
Browser("Practice Test Automation").Page("Login").WebButton("Submit").Click

'---Verify new page URL contains practicetestautomation.com/logged-in-successfully/
urlValue = Browser("Practice Test Automation").Page("Homepage").GetRoProperty("URL")

If urlValue = "practicetestautomation.com/logged-in-successfully/" Then
	Reporter.ReportEvent micPass, "URL validation","URL is correct"
Else
	Reporter.ReportEvent micFail, "URL validation","URL is NOT correct"
End If

'---Verify new page contains expected text ('Congratulations' or 'successfully logged in')
If Browser("Practice Test Automation").Page("Homepage").WebElement("Logged In Successfully").Exist AND Browser("Practice Test Automation").Page("Homepage").WebElement("Congratulations student. You successfully logged in!").Exist Then
	Reporter.ReportEvent micPass, "Homepage validation","Page contains expected text"
Else
	Reporter.ReportEvent micFail, "Homepage validation","Page DOESN'T contains expected text"
End If

'---Verify button Log out is displayed on the new page
If Browser("Practice Test Automation").Page("Homepage").WebButton("Log out").Exist Then
	Reporter.ReportEvent micPass, "Homepage validation","Log out button is displayed"
Else
	Reporter.ReportEvent micFail, "Homepage validation","Log out button is NOT displayed"
End If
